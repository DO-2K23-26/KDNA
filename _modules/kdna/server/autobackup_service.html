def create_dic_autobackup(data):
    li = []
    for ligne in data:
        dic = {
            "id": ligne[0].strip(),
            "frequency": ligne[1].strip(),
            "name": ligne[2].strip(),
            "timestamp": ligne[3].strip(),
            "id_server": ligne[4].strip(),
            "path": ligne[5].strip()
        }
        li.append(dic)
    return li
